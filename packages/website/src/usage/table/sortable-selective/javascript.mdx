```html
<admiralty-table caption="Favourite Foods">
  <admiralty-table-header>
    <admiralty-table-row>
      <admiralty-table-header-cell sortable="true" data-col="0">Name</admiralty-table-header-cell>
      <admiralty-table-header-cell sortable="true" data-col="1">Age</admiralty-table-header-cell>
      <admiralty-table-header-cell data-col="2">Other</admiralty-table-header-cell>
      <admiralty-table-header-cell data-col="3">Favourite Food</admiralty-table-header-cell>
    </admiralty-table-row>
  </admiralty-table-header>
  <admiralty-table-body id="sortable-selective-table-body"></admiralty-table-body>
</admiralty-table>

<script>
  const rows = [
    ["Tom", "12", "Likes Bread", "Potatoes"],
    ["Thomas", "25", "Likes Lots Of Bread", "Pizza"],
    ["Tommy", "1000", "Bread", "Potatoes"],
  ];

  const body = document.querySelector("#sortable-selective-table-body");
  const sortableCells = document.querySelectorAll("admiralty-table-header-cell[sortable='true']");

  const renderRows = (nextRows) => {
    body.innerHTML = nextRows
      .map(
        (row) =>
          `<admiralty-table-row>${row
            .map((cell) => `<admiralty-table-cell>${cell}</admiralty-table-cell>`)
            .join("")}</admiralty-table-row>`,
      )
      .join("");
  };

  renderRows(rows);

  sortableCells.forEach((cell) => {
    cell.addEventListener("admiraltySortChange", (event) => {
      const col = Number(cell.getAttribute("data-col"));
      const direction = event.detail.direction;
      const sortedRows = [...rows];

      sortableCells.forEach((other) => {
        if (other !== cell) {
          other.sortDirection = "none";
        }
      });

      if (direction === "ascending") {
        sortedRows.sort((a, b) => a[col].localeCompare(b[col], undefined, { numeric: true }));
      } else if (direction === "descending") {
        sortedRows.sort((a, b) => b[col].localeCompare(a[col], undefined, { numeric: true }));
      }

      renderRows(direction === "none" ? rows : sortedRows);
    });
  });
</script>
```
