```tsx
import React, { useMemo, useState } from "react";
import {
  AdmiraltyTable,
  AdmiraltyTableBody,
  AdmiraltyTableCell,
  AdmiraltyTableHeader,
  AdmiraltyTableHeaderCell,
  AdmiraltyTableRow,
} from "@ukho/admiralty-react";

const initialRows = [
  ["Tom", "12", "Likes Bread", "Potatoes"],
  ["Thomas", "25", "Likes Lots Of Bread", "Pizza"],
  ["Tommy", "1000", "Bread", "Potatoes"],
] as const;

type SortDirection = "none" | "ascending" | "descending";

export function SortableTableExample() {
  const [activeColumn, setActiveColumn] = useState<number | null>(null);
  const [activeDirection, setActiveDirection] = useState<SortDirection>("none");

  const rows = useMemo(() => {
    if (activeColumn === null || activeDirection === "none") {
      return initialRows;
    }

    const sorted = [...initialRows];
    sorted.sort((a, b) => {
      const result = a[activeColumn].localeCompare(b[activeColumn], undefined, { numeric: true });
      return activeDirection === "ascending" ? result : -result;
    });
    return sorted;
  }, [activeColumn, activeDirection]);

  const onSort = (column: number) => (event: CustomEvent<{ direction: SortDirection }>) => {
    if (event.detail.direction === "none") {
      setActiveColumn(null);
      setActiveDirection("none");
      return;
    }

    setActiveColumn(column);
    setActiveDirection(event.detail.direction);
  };

  return (
    <AdmiraltyTable caption="Favourite Foods" allow-sorting={true}>
      <AdmiraltyTableHeader>
        <AdmiraltyTableRow>
          <AdmiraltyTableHeaderCell
            sortable={true}
            sort-direction={activeColumn === 0 ? activeDirection : "none"}
            onAdmiraltySortChange={onSort(0)}>
            Name
          </AdmiraltyTableHeaderCell>
          <AdmiraltyTableHeaderCell
            sortable={true}
            sort-direction={activeColumn === 1 ? activeDirection : "none"}
            onAdmiraltySortChange={onSort(1)}>
            Age
          </AdmiraltyTableHeaderCell>
          <AdmiraltyTableHeaderCell
            sortable={true}
            sort-direction={activeColumn === 2 ? activeDirection : "none"}
            onAdmiraltySortChange={onSort(2)}>
            Other
          </AdmiraltyTableHeaderCell>
          <AdmiraltyTableHeaderCell
            sortable={true}
            sort-direction={activeColumn === 3 ? activeDirection : "none"}
            onAdmiraltySortChange={onSort(3)}>
            Favourite Food
          </AdmiraltyTableHeaderCell>
        </AdmiraltyTableRow>
      </AdmiraltyTableHeader>
      <AdmiraltyTableBody>
        {rows.map((row) => (
          <AdmiraltyTableRow key={`${row[0]}-${row[1]}`}>
            <AdmiraltyTableCell>{row[0]}</AdmiraltyTableCell>
            <AdmiraltyTableCell>{row[1]}</AdmiraltyTableCell>
            <AdmiraltyTableCell>{row[2]}</AdmiraltyTableCell>
            <AdmiraltyTableCell>{row[3]}</AdmiraltyTableCell>
          </AdmiraltyTableRow>
        ))}
      </AdmiraltyTableBody>
    </AdmiraltyTable>
  );
}
```
